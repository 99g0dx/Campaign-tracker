. Backend: include status in the public share API

Update your public campaign route so it returns posts (with status) as well as the campaign.

// server/routes/publicCampaign.ts
import { campaigns, posts } from "../db/schema";
import { eq } from "drizzle-orm";

router.get("/:slug", async (req, res) => {
  const { slug } = req.params;
  const cookieName = COOKIE_PREFIX + slug;

  if (req.cookies?.[cookieName] !== "ok") {
    return res.status(401).json({ error: "Locked" });
  }

  const [campaign] = await db
    .select()
    .from(campaigns)
    .where(eq(campaigns.shareSlug, slug));

  if (!campaign || !campaign.shareEnabled) {
    return res.status(404).json({ error: "Not found" });
  }

  const creatorPosts = await db
    .select({
      id: posts.id,
      creatorName: posts.creatorName,
      platform: posts.platform,
      status: posts.status,          // ðŸ‘ˆ this is what we need
      url: posts.url,
      views: posts.views,
      likes: posts.likes,
      comments: posts.comments,
      shares: posts.shares,
    })
    .from(posts)
    .where(eq(posts.campaignId, campaign.id));

  res.json({ campaign, creatorPosts });
});


So the shared endpoint now gives you creatorPosts with a status field.

2. Frontend: show status on the shared dashboard

In your SharedCampaignPage, when you fetch the public data, also store creatorPosts:

// client/src/pages/SharedCampaignPage.tsx
type CreatorPost = {
  id: string;
  creatorName: string;
  platform: string;
  status: "active" | "pending" | "briefed" | "completed" | string;
  url?: string;
  views: number;
  likes: number;
  comments: number;
  shares: number;
};

const [creatorPosts, setCreatorPosts] = useState<CreatorPost[]>([]);

async function loadCampaign(slug: string) {
  const res = await fetch(`/api/public/campaigns/${slug}`);
  if (!res.ok) return;
  const data = await res.json();
  setCampaign(data.campaign);
  setCreatorPosts(data.creatorPosts || []);
  setUnlocked(true);
}


Call loadCampaign(slug) both after verify and in the initial useEffect.

Then, under your engagement chart on the shared page, render a read-only Creators & Posts table with status badges:

function StatusBadge({ status }: { status: string }) {
  const normalized = status.toLowerCase();
  const label =
    normalized === "active"
      ? "Active"
      : normalized === "pending"
      ? "Pending"
      : normalized === "briefed"
      ? "Briefed"
      : normalized === "completed"
      ? "Completed"
      : status;

  const bg =
    normalized === "active"
      ? "bg-emerald-500/15 text-emerald-300 border-emerald-500/40"
      : normalized === "completed"
      ? "bg-sky-500/15 text-sky-300 border-sky-500/40"
      : normalized === "briefed"
      ? "bg-amber-500/15 text-amber-300 border-amber-500/40"
      : "bg-slate-700/40 text-slate-200 border-slate-500/40";

  return (
    <span
      className={`inline-flex items-center px-2 py-0.5 rounded-full border text-[10px] font-medium ${bg}`}
    >
      {label}
    </span>
  );
}
